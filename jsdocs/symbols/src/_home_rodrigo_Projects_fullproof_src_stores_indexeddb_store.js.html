 29 	function install_on_request(req, success, error) {
 30 		req.onsuccess = success;
 31 		req.onerror = error;
 32 		return req;
 33 	}

 36 		if (tx.db.objectStoreNames.contains(name)) {
 37 			return tx.objectStore(name);
 38 		} else {
 39 			return tx.db.createObjectStore(name, parameter);

 43 	function setObject(store, object, callback, error) {
 44 		var req = store.put(object);
 45 		install_on_request(req, fullproof.make_callback(callback, object), error);
 46 	}

 48 	function getOrCreateObject(store, keyValue, defaultValue, callback, error) {
 49 		
 50 		function create() {
 51 			if (fullproof.isFunction(defaultValue)) {
 52 				defaultValue = defaultValue();
 53 			}
 54 			setObject(store, defaultValue, callback, error);
 55 		}
 56 		try {
 57 			var req = store.get(keyValue);
 58 		} catch (e) {
 59 			console && console.log && console.log(e);
 60 		}
 61 
 62 		req.onsuccess = function(ev) {
 63 			if (ev.target.result === undefined) {
 64 				if (defaultValue !== undefined) {
 65 					create();
 66 				} else {
 67 					error();
 68 				}
 69 			} else {
 70 				callback(req.result);
 71 			}
 72 		};
 73 		req.onerror = create;
 74 	};

 84 	function IndexedDBIndex(parent, database, indexName, comparatorObject, useScore) {
 85 		this.parent = parent;
 86 		this.database = database;
 87 		this.name = indexName;
 88 		this.comparatorObject = comparatorObject;
 89 		this.useScore = useScore;
 90 		this.internalComparator = useScore?function(a,b) {
 91 			return this.comparatorObject(a.value,b.value);
 92 		}:function(a,b) {
 93 			return this.comparatorObject(a,b);
 94 		};
 95 	}

 97 	IndexedDBIndex.prototype.clear = function (callback) {
 98         callback = callback || function(){};
 99         var self = this;
100         var wrongfunc = fullproof.make_callback(callback, false);
101         var tx = this.database.transaction([this.name, this.parent.metaStoreName], fullproof.store.READWRITEMODE);
102         var metastore = tx.objectStore(this.parent.metaStoreName);
103         install_on_request(metastore.put({id:this.name, init:false}), function () {
104             fullproof.call_new_thread(function () {
105                 var tx = self.database.transaction([self.name], fullproof.store.READWRITEMODE);
106                 var store = tx.objectStore(self.name);
107                 var req = store.clear();
108                 install_on_request(req, fullproof.make_callback(callback, true), wrongfunc);
109             });
110         }, wrongfunc);
111     };

113 	IndexedDBIndex.prototype.inject = function(word, value, callback) {
114 		var tx = this.database.transaction([this.name], fullproof.store.READWRITEMODE);
115 		var store = tx.objectStore(this.name);
116 		var self = this;
117 		getOrCreateObject(store, word, function() { return {key:word,v:[]} }, function(obj) {
118 			var rs = new fullproof.ResultSet(self.comparatorObject).setDataUnsafe(obj.v);
119 			if (value instanceof fullproof.ScoredElement) {
120 				if (self.useScore) {
121 					rs.insert({v:value.value, s:value.score});
122 				} else {
123 					rs.insert(value.value);
124 				}
125 			} else {
126 				rs.insert(value);
127 			}
128 			obj.v = rs.getDataUnsafe();
129 			setObject(store, obj, callback, fullproof.make_callback(callback, false));
130 		}, fullproof.make_callback(callback,false));
131 	};

133     var storedObjectComparator_score = {
134         lower_than: function(a,b) {
135             return (a.v?a.v:a)<(b.v? b.v:b);
136         },
137         equals: function(a,b) {
138             return (a.v?a.v:a)==(b.v? b.v:b);
139         }

142     function createMapOfWordsToResultSet(self, wordArray, valuesArray, offset, count, resultPropertiesAsArray) {
143 		var result = new fullproof.HMap();
144 		for (; offset < count; ++offset) {
145 			var word = wordArray[offset];
146 			var value = valuesArray[offset];
147 
148 			if (result.get(word) === undefined) {
149 				result.put(word, new fullproof.ResultSet(storedObjectComparator_score));
150                 resultPropertiesAsArray.push(word);
151             }
152 			if (value instanceof fullproof.ScoredElement) {
153 				if (self.useScore) {
154                     var rs = result.get(word);
155                     rs.insert({v:value.value, s:value.score});
156 				} else {
157                     result.get(word).insert(value.value);
158 				}
159 			} else {
160                 result.get(word).insert(value);
161 			}
162 		}
163 		return result;
164 	}

166 	function storeMapOfWords(self, store, words, data, callback, offset, max) {
167 		if (words.length>0 && offset < max) {
168 			var word = words[offset];
169 			var value = data.get(word);
170 			getOrCreateObject(store, word, function() { return {key:word,v:[]} }, function(obj) {
171 				var rs = new fullproof.ResultSet(self.comparatorObject).setDataUnsafe(obj.v);
172 				rs.merge(value);
173 				obj.v = rs.getDataUnsafe();
174 				setObject(store, obj, function() {
175 					storeMapOfWords(self, store, words, data, callback, offset+1, max);
176 				}, function() { callback(false); });
177 			});
178 		} else {
179 			callback(true);
180 		}
181 	}

183 	IndexedDBIndex.prototype.injectBulk = function (wordArray, valuesArray, callback, progress) {
184         var self = this;
185         if (wordArray.length != valuesArray.length) {
186             throw "Can't injectBulk, arrays length mismatch";
187         }
188 
189         var self = this;
190         var batchSize = 100;
191 
192         var words = [];
193         var data = createMapOfWordsToResultSet(this, wordArray, valuesArray, 0, wordArray.length, words);
194 
195         function storeData(self, words, data, callback, progress, offset) {
196             if (progress) {
197                 progress(offset / words.length);
198             }
199             var tx = self.database.transaction([self.name], fullproof.store.READWRITEMODE);
200             var store = tx.objectStore(self.name);
201             storeMapOfWords(self, store, words, data, function (res) {
202                 if (offset < words.length) {
203                     fullproof.call_new_thread(storeData, self, words, data, callback, progress, offset + batchSize)
204                 } else {
205                     callback(res);
206                 }
207             }, offset, Math.min(offset + batchSize, words.length));
208         }
209 
210         if (words.length > 0) {
211             storeData(this, words, data, callback, progress, 0)
212         } else {
213             callback(true);
214         }
215     };

217 	IndexedDBIndex.prototype.lookup = function(word, callback) {
218 		var tx = this.database.transaction([this.name]);
219 		var store = tx.objectStore(this.name);
220 		var self = this;
221 		getOrCreateObject(store, word, undefined, function(obj) {
222 			if (obj && obj.v) {
223 				var rs = new fullproof.ResultSet(self.comparatorObject);
224 				for (var i=0,max=obj.v.length; i<max; ++i) {
225 					var o = obj.v[i];
226 					if (self.useScore) {
227 						rs.insert(new fullproof.ScoredEntry(word, o.v, o.s));
228 					} else {
229 						rs.insert(o);
230 					}
231 				}
232 				callback(rs);
233 			} else {
234 				callback(new fullproof.ResultSet(self.comparatorObject));
235 			}
236 		}, function() { callback(new fullproof.ResultSet(self.comparatorObject)); });
237 	};

272 	function createStores(database, indexRequestArray, metaStoreName) {
273 		if (!database.objectStoreNames.contains(metaStoreName)) {
274 			database.createObjectStore(metaStoreName, {keyPath: "id"});
275 		}
276 		for (var i=0; i<indexRequestArray.length; ++i) {
277 			if (!database.objectStoreNames.contains(indexRequestArray[i].name)) {
278 				database.createObjectStore(indexRequestArray[i].name, {keyPath: "key"});
279 			}
280 		}

296 		function setupIndexes(self) {
297 			for (var i=0; i<self.indexRequests.length; ++i) {
298 				var ireq =self.indexRequests[i];
299 				var compObj = ireq.capabilities.getComparatorObject()?ireq.capabilities.getComparatorObject():(self.useScore?fullproof.ScoredElement.comparatorObject:undefined);
300 				var index = new IndexedDBIndex(self, self.database, ireq.name, compObj, ireq.capabilities.getUseScores());
301 				self.stores[ireq.name] = index;

306 		function callInitializerIfNeeded(database, self, indexRequestArray, callback, errorCallback) {
307 			if (indexRequestArray.length == 0) {
308 				return callback(true);
309 			}
310 			
311 			var tx = database.transaction([self.metaStoreName], fullproof.store.READWRITEMODE);
312 			var metastore = tx.objectStore(self.metaStoreName);
313 			var ireq = indexRequestArray.shift();
314             getOrCreateObject(metastore, ireq.name, {id: ireq.name, init: false},
315 					function(obj) {
316 						if (obj.init == false && ireq.initializer) {
317 							var initIndex = self.getIndex(ireq.name);
318 							fullproof.call_new_thread(function() {
319 								initIndex.clear(function() {
320 									ireq.initializer(self.getIndex(ireq.name), function() {
321 										fullproof.call_new_thread(callInitializerIfNeeded, database, self, indexRequestArray, callback, errorCallback);
322 										fullproof.call_new_thread(function() {
323 											var tx = database.transaction([self.metaStoreName], fullproof.store.READWRITEMODE);
324 											var metastore = tx.objectStore(self.metaStoreName);
325 											obj.init = true;
326 											install_on_request(metastore.put(obj), function(){}, function(){});;
327 										});
328 									});
329 								});
330 							});
331 						} else {
332 							fullproof.call_new_thread(callInitializerIfNeeded, database, self, indexRequestArray, callback, errorCallback);
333 						}
334 					}, errorCallback);
335 		}

337 		function checkInit(self, database, indexRequestArray, callback, errorCallback) {
338 			createStores(database, indexRequestArray, self.metaStoreName);
339 			setupIndexes(self);
340 			// callInitializerIfNeeded(database, self, [].concat(indexRequestArray), callback, errorCallback);
341             fullproof.call_new_thread(callInitializerIfNeeded, database, self, [].concat(indexRequestArray), callback, errorCallback);
342 		}
343 		
344 		this.indexRequests = reqIndexArray;
345 		var self = this;
346 		
347 		var openRequest = fullproof.store.indexedDB.open(this.dbName, this.dbVersion);
348 		openRequest.onerror = function() { 
349 			errorCallback() 
350 		};
351 		openRequest.onsuccess = function(ev) {
352 			self.database = ev.result || ev.target.result;
353 					
354 			if (self.database.version !== undefined && self.database.setVersion && self.database.version != self.dbVersion) {
355 				var versionreq = self.database.setVersion(self.dbVersion);
356 				versionreq.onerror = fullproof.make_callback(errorCallback, "Can't change version with setVersion(" +self.dbVersion+")");
357 				versionreq.onsuccess = function(ev) {
358 					createStores(self.database, reqIndexArray, self.metaStoreName);
359 					checkInit(self, self.database, self.indexRequests, 
360 							function() { 
361 								callback(indexArrayResult); 
362 							}, errorCallback);
363 				}
364 			} else {
365 				checkInit(self, self.database, self.indexRequests, fullproof.make_callback(callback, indexArrayResult), errorCallback);
366 			}	
367 		};
368 		openRequest.onupgradeneeded = function(ev) {
369 			createStores(ev.target.result, reqIndexArray, self.metaStoreName);
370 			updated = true;
371 		};
372 		
373 	};
