 19 /**
 20  * An object that associates a value and a numerical score
 21  * @constructor
 22  */
 23 fullproof.ScoredElement = function (value, score) {
 24     if (!(this instanceof fullproof.ScoredElement)) {
 25         return new fullproof.ScoredElement(value, score);
 26     }
 27     this.value = value;
 28     this.score = score === undefined ? 1.0 : score;
 29 };
 30 
 31 fullproof.ScoredElement.prototype.toString = function() {
 32 	return "["+this.value+"|"+this.score+"]";
 33 };
 34 fullproof.ScoredElement.prototype.getValue = function () {
 35     return this.value;
 36 };
 37 fullproof.ScoredElement.prototype.getScore = function () {
 38     return this.score;
 39 };

 77 /**
 78  * Creates a synchronization point. This function returns a function that collects
 79  * calls to callback, then calls its callback argument with all the data collected.
 80  * The synchronization point can trigger the final call to callback when it either
 81  * receives a fixed number of calls (expected argument >= 1), or when it
 82  * receives a false boolean value as argument (expected has to be either undefined or false)
 83 
 84  * @param {function} callback the function to call when the synchronization point is reached
 85  * @param {number} expected defines the synchronization point. If this is a number, the synchronization is
 86  * triggered when the function returned is called this number of times. If this is set undefined, the sync is
 87  * triggered when this function returned is called with a single argument {boolean} false.
 88  * @param debug if defined, some debugging information are printed to the console, if it exists.
 89  */
 90 fullproof.make_synchro_point = function (callback, expected, debug, thrown_if_false) {
 91     var count = 0;
 92     var results = [];
 93     return function (res) {
 94         if (thrown_if_false !== undefined && res === false) {
 95             throw thrown_if_false;
 96         }
 97         if (expected === false || expected === undefined) {
 98             if (res === false) {
 99                 callback(results);
100             } else {
101                 results.push(res);
102             }
103         } else {
104 
105             ++count;
106             results.push(res);
107             if (debug && console && console.log) {
108                 console.log("synchro point " + (typeof debug == "string" ? debug + ": " : ": ") + count + " / " + expected);
109             }
110             if (count == expected) {
111                 callback(results);
112             }
113         }
114     }
115 };

117 fullproof.call_new_thread = function() {
118 	var args = Array.prototype.slice.call(arguments);
119 	setTimeout(function() {
120 		if (args.length>0) {
121 			var func = args.shift();
122 			func.apply(this, args);
123 		}
124 	}, 1);
125 };

127 /**
128  * Creates and returns a function that, when called, calls the callback argument, with any number
129  * of arguments.
130  * @param {function} callback a function reference to call when the created function is called
131  * @param {...*} varargs any number of arguments that will be applied to the callback function, when called.
132  */
133 fullproof.make_callback = function(callback) {
134 	var args = Array.prototype.slice.call(arguments, 1);
135 	return function() {
136 		if (callback) {
137 			callback.apply(this, args);
138 		}
139 	}
140 };

155 fullproof.filterObjectProperties = function (array_of_object, property) {
156     if (array_of_object instanceof fullproof.ResultSet) {
157         array_of_object = array_of_object.getDataUnsafe();
158     }
159     var result = [];
160     for (var i = 0, max = array_of_object.length; i < max; ++i) {
161         result.push(array_of_object[i][property]);
162     }
163     return result;
164 };

185 /**
186  * An HashMap structure that uses a javascript object to store its data, and prefixes all the keys
187  * with a '$' to avoid name conflict with object properties.
188  * @constructor
189  */
190 fullproof.HMap = function() {
191 }
192 
193 fullproof.HMap.prototype.put = function(k,v) {
194     this["$"+k] = v;
195 }
196 fullproof.HMap.prototype.putInArray = function(k,v) {
197     var $k = "$"+k;
198     if (!this[$k] || this[$k].constructor !== Array) {
199         this[$k] = [];
200     }
201     this[$k].push(v);
202 }
203 fullproof.HMap.prototype.get = function(k) {
204     return this["$"+k];
205 }
206 fullproof.HMap.prototype.forEach = function(func) {
207     for (var k in this) {
208        if ("$" === k[0]) {
209            func(k.substring(1));
210        }
211     }
212 }
