 19     /**
 20      * Starts the MameSearch search engine

 22      * @param progress if the indexes need to be initialized, this function is called with the progress value, ranging from 0 to 1.
 23      */
 24     this.start = function(callback, progress) {
 25 
 26         var loader = new fullproof.DataLoader();
 27 		loader.setQueue("mamegames.txt");
 28 		loader.start(function() {
 29 
 30             var index1 = new fullproof.IndexUnit("normalindex",
 31                 new fullproof.Capabilities().setStoreObjects(false).setUseScores(true).setDbName(dbName).setComparatorObject(fullproof.ScoredEntry.comparatorObject),
 32                 new fullproof.ScoringAnalyzer(fullproof.normalizer.to_lowercase_nomark, fullproof.normalizer.remove_duplicate_letters),
 33                 makeInitializer(function(val) { progress(val/2); }));
 34 
 35             var index2 = new fullproof.IndexUnit("stemmedindex",
 36                 new fullproof.Capabilities().setStoreObjects(false).setUseScores(true).setDbName(dbName).setComparatorObject(fullproof.ScoredEntry.comparatorObject),
 37                 new fullproof.ScoringAnalyzer(fullproof.normalizer.to_lowercase_nomark, fullproof.english.metaphone),
 38                 makeInitializer(function(val){progress(val/2 + 0.5);}));
 39 
 40             engine.open([index1, index2], fullproof.make_callback(callback, true), fullproof.make_callback(callback, false));
 41 
 42 		}, function(txt, file) { data = txt.split("\n"); }, 
 43 		 function() { console.log("ERROR");});
 44 		
 45 	}

 47 	this.lookup = function(txt, callback) {
 48 		engine.lookup(txt, function(resultset) {
 49 
 50 			if (!resultset) {
 51 				return "no match.";
 52 			}
 53 			
 54 			resultset.setComparatorObject({
 55 				lower_than: function(a,b) {
 56 					return a.score > b.score;
 57 				},
 58 				equals: function(a,b) {
 59 					return a.score === b.score;
 60 				}
 61 			});
 62 			
 63 			var result = "";
 64 			if (resultset.getSize() == 0) {
 65 				result += "<div style='font-weight:bold;'>No result found for query '" + txt + '"</div>';
 66 			} else {
 67 				result += "<div>" + resultset.getSize() + " entr" + (resultset.getSize()>1?"ies were":"y was") + " found</div>";
 68 			}
 69 			result += "<table><tr><th>ROM</th><th>Name</th><th>Search score</th></tr>";
 70 			resultset.forEach(function(entry) {
 71 				if (entry instanceof fullproof.ScoredElement) {
 72 					var line = data[entry.value];
 73 					var split = line.split(";");
 74 					result += "<tr><td>" + (split[0]?split[0]:"-") +"</td><td>" + (split[1]?split[1]:"-") + "</td><td>"+entry.score.toFixed(3)+"</td></tr>";
 75 				} else {
 76 					var split = data[line].split(";");
 77 					result += "<tr><td>" + (split[0]?split[0]:"-") +"</td><td>" + (split[1]?split[1]:"-") + "</td></tr>"; 
 78 				}
 79 			});
 80 			result += "</table>";
 81 			
 82 			callback(result);
 83 		});
 84 	};
