  3 function init_store(store, capabilities, indexReqArray, indexName, injectScoredEntryResultSet, callback) {
  4 	store.close(function() {
  5 		store.open(capabilities, indexReqArray, function() {
  6 			if (indexName) {
  7 				var index = store.getIndex(indexName);
  8 				index.clear(function() {
  9 					if (injectScoredEntryResultSet) {
 10 						var synchro_inject = fullproof.make_synchro_point(fullproof.make_callback(callback, index), injectScoredEntryResultSet.getSize());
 11 						injectScoredEntryResultSet.forEach(function(el) {
 12 							index.inject(el.key, injectScoredEntryResultSet.useScore?new fullproof.ScoredElement(el.value, el.score):el.value, synchro_inject);
 13 						});
 14 					} else {
 15 						callback(index);
 16 					}
 17 				});
 18 			} else {
 19 				callback();
 20 			}
 21 		}, fullproof.thrower("Can't open store"));
 22 	});
 23 }

181 function mkDataGenerator(integerValueOnly) {
182 	return function(count, overlap) {
183 		var set1 = integerValueOnly?fullproof.tests.makeResultSetOfScoredEntries(count,100*count):fullproof.tests.makeResultSetOfScoredEntriesObjects(count);
184 		var set2 = integerValueOnly?fullproof.tests.makeResultSetOfScoredEntries(overlap?count/2:count,100*count):fullproof.tests.makeResultSetOfScoredEntriesObjects(overlap?count/2:count);
185 		var set3 = new ResultSet;
186 		if (overlap){
187 			set3 = integerValueOnly?fullproof.tests.makeResultSetOfScoredEntries(overlap?count/2:count,100*count):fullproof.tests.makeResultSetOfScoredEntriesObjects(overlap?count/2:count);
188 			set2.merge(set3);
189 		}
190 		var merged = new fullproof.ResultSet().merge(set1,set2);
191 		var expected = [];
192 		merged.forEach(function(e) {expected.push([e]);});
193 		return {
194 			set1: set1, 
195 			set2: set2,
196 			expected: expected
197 		}
198 	}
