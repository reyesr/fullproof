 32 	function sql_table_exists_or_empty(tx, tablename, callback) {
 33 		tx.executeSql("SELECT * FROM " + tablename + " LIMIT 1,0", [], function(tx,res) {
 34 			if (res.rows.length == 1) {
 35 				callback(true);
 36 			} else {
 37 				callback(false);
 38 			}
 39 		}, fullproof.make_callback(callback, false));

 46 		this.createIndex = function(name, successCallback, errorCallback) {
 47 			var error = fullproof.make_callback(errorCallback);

 49 			store.db.transaction(function(tx) {

 51 					if (!exists) {
 52 						tx.executeSql("CREATE TABLE IF NOT EXISTS "+ name +" (id NCHAR(48), value, score)", [], 
 53 									function() {
 54 										store.db.transaction(function(tx) {				
 55 											tx.executeSql("CREATE INDEX IF NOT EXISTS "+ name +"_indx ON " + name + " (id)", [], function() {

 58 														error);
 59 											}, error);
 60 										});
 61 									}, error);
 62 					} else {

174 	function openIndex(store, name, parameters, initializer, callback, errorCallback) {
175 		if (store.opened == false || !store.meta) {
176 			return callback(false);
177 		}
178 		
179 		parameters = parameters||{};
180 		var index = new WebSQLStoreIndex();
181 		index.store = store;
182 		var useScore = parameters.getUseScores()!==undefined?(parameters.getUseScores()):false;
183 		
184 		index.db = store.db;
185 		index.tableName = index.name = name;
186 		index.comparatorObject = parameters.getComparatorObject()?parameters.getComparatorObject():(useScore?fullproof.ScoredElement.comparatorObject:undefined);
187 		index.useScore = useScore;
188 		
189 		var self = store;
190 		store.meta.isInitialized(name, function(isInit) {
191 			if (isInit) {
192 				return callback(index);
193 			} else {
194 				self.meta.createIndex(name, function() {
195 					self.indexes[name] = index;
196 					if (initializer) {
197                         fullproof.call_new_thread(function() {
198                             index.clear(function() {
199                                 fullproof.call_new_thread(function() {
200                                     initializer(index, function() {
201                                         index.opened = true;
202                                         self.meta.setInitialized(name, true, fullproof.make_callback(callback, index));
203                                     });
204                                 });
205                             });
206                         });
207 					} else {
208 						callback(index);
209 					}
210 				}, errorCallback);
211 			}
212 		});				
213 	};

215 	function openStore(store, parameters, callback) {
216 		store.opened = false;
217 		if (parameters.getDbName() !== undefined) {
218 			store.dbName = parameters.getDbName();
219 		}
220 		if (parameters.getDbSize() !== undefined) {
221 			store.dbSize = parameters.getDbSize();
222 		}
223 		store.db = openDatabase(store.dbName, '1.0', 'javascript search engine', store.dbSize*10);
224 		store.opened = true;
225 		store.meta = new MetaData(store, function(store) {
226 				callback(store);
227 			}, fullproof.make_callback(callback,false));
228 	};

231 	fullproof.store.WebSQLStore.prototype.open = function(caps, reqIndexArray, callback, errorCallback) {
232 		var self = this;
233 		var resultArray = [];
234 		this.dbName = caps.getDbName() || this.dbName;
235 		function chainOpenIndex(reqIndexes) {
236 			if (reqIndexes.length == 0) {
237 				return callback(resultArray);
238 			}
239 			var requestIndex = reqIndexes.shift();
240 			openIndex(self, requestIndex.name, requestIndex.capabilities, requestIndex.initializer, function(index) {
241 				resultArray.push(index);
242 				chainOpenIndex(reqIndexes);
243 			});
244 		}
245 		
246 		openStore(this, caps, function(store) {
247 			var synchro = fullproof.make_synchro_point(callback, reqIndexArray.length);
248 			var consumedReqIndexes = [].concat(reqIndexArray);
249 			chainOpenIndex([].concat(reqIndexArray));
250 		});
251 	};

285 	WebSQLStoreIndex.prototype.injectBulk = function(wordArray, valuesArray, callback, progress) {
286 		var self = this;
287 		if (wordArray.length != valuesArray.length) {
288 			throw "Can't injectBulk, arrays length mismatch";
289 		}
290 		var batchSize = 100;
291 		var transactionsExpected = wordArray.length / batchSize + (wordArray%batchSize>0?1:0);
292 		var bulk_synchro = fullproof.make_synchro_point(callback, undefined, true);
293 		var totalSize = wordArray.length;
294 		var processBulk = function(wArray, vArray) {
295             console.log("injectBulk iteration, rem:     " + vArray.length);
296 			var curWords = wArray.splice(0, batchSize<wArray.length?batchSize:wArray.length);
297 			var curValues = vArray.splice(0, batchSize<vArray.length?batchSize:vArray.length);
298 			if (curWords.length == 0) {
299 				bulk_synchro(false);
300 			}
301 			if (progress) {
302 				progress((totalSize - wArray.length)/totalSize);
303 			}
304 			
305 			var synchronizer = fullproof.make_synchro_point(function() {
306 				fullproof.call_new_thread(processBulk, wArray, vArray);
307 			}, curWords.length);
308 			
309 			self.db.transaction(function(tx) {
310 				for (var i=0; i<curWords.length; ++i) {
311 					var value = curValues[i];
312 					if (value instanceof fullproof.ScoredEntry) {
313 						if (self.useScore) {
314 							tx.executeSql("INSERT INTO " + self.tableName + " (id,value, score) VALUES (?,?,?)", [curWords[i], value.value, value.score], synchronizer, function() {
315                                 console.log("ERROR SQL");
316                             });
317 						} else {
318 							tx.executeSql("INSERT INTO " + self.tableName + " (id,value) VALUES (?,?)", [curWords[i], value.value], synchronizer, synchronizer);
319 						}
320 					} else {
321 						if (self.useScore) {
322 							tx.executeSql("INSERT INTO " + self.tableName + " (id,value, score) VALUES (?,?,?)", [curWords[i], value, 1.0], synchronizer, synchronizer);
323 						}
324 						else {
325 							tx.executeSql("INSERT INTO " + self.tableName + " (id,value) VALUES (?,?)", [curWords[i], value], 
326 									function() {
327 								synchronizer();
328 							}, function() {
329 								synchronizer(true);
330 							});
331 						}
332 					}
333 				}
334 			});
335 		};
336 		
337 		processBulk(wordArray, valuesArray);
338 		
339 	};

341 	/**
342 	 * WebSQLStore does not support object storage, only primary values, so we rely
343 	 * on the sql engine sorting functions. ORDER BY should provide fine results as long as
344 	 * the datatype of values is consistant.
345 	 */
346 	WebSQLStoreIndex.prototype.lookup = function(word, callback) {
347 		var self = this;
348 		this.db.transaction(function(tx) {
349 			tx.executeSql("SELECT * FROM " + self.tableName + " WHERE id=? ORDER BY value ASC", [word],
350 					function(tx,res) {
351 						var result = [];
352 						for (var i=0; i<res.rows.length; ++i) {
353 							var item = res.rows.item(i);
354 							if (item) {
355 								if (item.score === null || item.score === undefined || item.score === false) {
356 									result.push(item.value);
357 								} else {
358 									result.push(new fullproof.ScoredEntry(item.id, item.value, item.score));
359 								}
360 							}
361 						}
362 						callback(new fullproof.ResultSet(self.comparatorObject).setDataUnsafe(result));
363 					}, 
364 					function() {
365 						callback(false);
366 					});
367 		});
