 19 /**
 20  * A boolean-set based engine.
 21  *
 22  * @constructor
 23  */
 24 fullproof.BooleanEngine = function (storeDescriptors) {
 25 
 26     if (!(this instanceof fullproof.BooleanEngine)) {
 27         return new fullproof.BooleanEngine(storeDescriptors);
 28     }
 29 
 30     this.initAbstractEngine(storeDescriptors); // Init from the prototype
 31 
 32     /**
 33      * The working mode when gathering result sets.
 34      */
 35     this.booleanMode = fullproof.BooleanEngine.CONST_MODE_INTERSECT;
 36 
 37 
 38 
 39     function lookup(self, text, callback, arrayOfIndexUnits, mode) {
 40         if (arrayOfIndexUnits.length == 0) {
 41             return callback(false);
 42         }
 43         var unit = arrayOfIndexUnits.shift();
 44         ++(self.lastResultIndex);
 45         unit.analyzer.parse(text, fullproof.make_synchro_point(function (array_of_words) {
 46 
 47             if (!array_of_words || array_of_words.length == 0) {
 48                 if (arrayOfIndexUnits.length > 0) {
 49                     return lookup(self, text, callback, arrayOfIndexUnits, mode);
 50                 } else {
 51                     return callback(false);
 52                 }
 53             }
 54 
 55             var lookup_synchro = fullproof.make_synchro_point(function (rset_array) {
 56 
 57                 var curset = rset_array.shift();
 58                 while (rset_array.length > 0) {
 59                     var set = rset_array.shift();
 60                     switch (mode) {
 61                         case fullproof.BooleanEngine.CONST_MODE_UNION:
 62                             curset.merge(set);
 63                             break;
 64                         default: // default is intersect
 65                             curset.intersect(set);
 66                             break;
 67                     }
 68                 }
 69 
 70                 if (curset.getSize() == 0) {
 71                     if (arrayOfIndexUnits.length > 0) {
 72                         return lookup(self, text, callback, arrayOfIndexUnits, mode);
 73                     } else {
 74                         callback(false);
 75                     }
 76                 } else {
 77                     callback(curset);
 78                 }
 79 
 80             }, array_of_words.length);
 81 
 82             for (var i = 0; i < array_of_words.length; ++i) {
 83                 unit.index.lookup(array_of_words[i], lookup_synchro);
 84             }
 85         }));
 86     }
 87 
 88     this.lookup = function (text, callback) {
 89         this.lastResultIndex = 0;
 90         lookup(this, text, callback, this.getIndexUnits(), this.booleanMode);
 91         return this;
 92     }
 93 };
