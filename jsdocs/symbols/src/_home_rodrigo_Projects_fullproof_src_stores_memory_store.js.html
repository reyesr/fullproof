 51 	function openIndex(store, name, parameters, initializer, callback) {
 52 		var index = new MemoryStoreIndex();
 53 		var useScore = parameters.getUseScores()!==undefined?(parameters.getUseScores()):false;
 54 		index.comparatorObject = parameters.getComparatorObject()?parameters.getComparatorObject():(useScore?fullproof.ScoredElement.comparatorObject:undefined);
 55 		index.useScore = useScore;
 56         index.name = name;
 57 		store.indexes[name] = index;
 58 		if (initializer) {
 59 			initializer(index, function() {
 60 				callback(index);
 61 			});
 62 		} else {
 63 			callback(index);
 64 		}
 65 		return index;
 66 	};

 68 	fullproof.store.MemoryStore.prototype.open = function(caps, reqIndexArray, callback, errorCallback) {
 69 		var self = this;
 70 		openStore(caps, function(store) {
 71 			var synchro = fullproof.make_synchro_point(function(result) {
 72 				callback(result);
 73 			}, reqIndexArray.length);
 74 			for (var i=0, max=reqIndexArray.length; i<max; ++i) {
 75 				var requestIndex = reqIndexArray[i];
 76 				openIndex(self, requestIndex.name, requestIndex.capabilities, requestIndex.initializer, synchro);
 77 			}
 78 		});
 79 	};

 99 	/**
100 	 * Inject data. Can be called as follows:
101 	 * memstoreInstance.inject("someword", 31321321, callbackWhenDone);
102 	 * memstoreInstance.inject("someword", new fullproof.ScoredElement(31321321, 1.0), callbackWhenDone);
103 	 * 
104 	 * When score is not set, and store is configured to store a score, then it is saved as undefined.
105 	 * When the score is set, and the store is configured not to store a score, it raises an exception
106 	 */
107 
108 	MemoryStoreIndex.prototype.inject = function(key, value, callback) {
109 		if (!this.data[key]) {
110 			this.data[key] = new fullproof.ResultSet(this.comparatorObject);
111 		}
112 		if (this.useScore === false && value instanceof fullproof.ScoredElement) {
113 			this.data[key].insert(value.value);
114 		} else {
115 			this.data[key].insert(value);
116 		}
117 
118 		if (callback) {
119 			callback(key,value);
120 		}
121 		
122 		return this;
123 	};

125 	MemoryStoreIndex.prototype.injectBulk = function(keyArray, valueArray, callback, progress) {
126 		for (var i=0; i<keyArray.length && i<valueArray.length; ++i) {
127             if (i%1000 == 0 && progress) {
128                 progress(i / keyArray.length);
129 
130             }
131 			this.inject(keyArray[i], valueArray[i]);
132 		}
133 		if (callback) {
134 			callback(keyArray,valueArray);
135 		}
136 		return this;
137 	};
