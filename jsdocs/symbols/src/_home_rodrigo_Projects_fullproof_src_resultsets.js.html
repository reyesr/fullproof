 22 	var defaultComparator = {
 23 		lower_than: function(a,b) {
 24 			return a<b;
 25 		},
 26 		equals: function(a,b) {
 27 			return a==b;
 28 		}

 32 	/**
 33      * Binary search an array.
 34      * @param array an array to search in
 35      * @param value the value to search for
 36      * @param min the floor index
 37      * @param max the ceil index
 38      * @param lower_than an optional function that takes two arguments and returns true if
 39      * the first argument is lower than the second one. You have to provide this function
 40      * if the values stored in the array cannot be sorted by the default javascript < operator.
 41      *
 42      * @return the index of the value, if found, or the index where the value can be inserted if not found.
 43      */
 44     fullproof.binary_search = function (array, value, min, max, lower_than) {
 45         lower_than = lower_than || defaultComparator.lower_than;
 46         if (min === undefined && max === undefined) {
 47             if (array.length == 0) {
 48                 return 0
 49             } else {
 50                 return fullproof.binary_search(array, value, 0, array.length, lower_than);
 51             }
 52         }
 53 
 54         while (max >= min) {
 55             var mid = parseInt((max + min) / 2);
 56             if (mid >= array.length) {
 57                 return array.length;
 58             } else if (lower_than(array[mid], value)) {
 59                 min = mid + 1;
 60             } else if (lower_than(value, array[mid])) {
 61                 max = mid - 1;
 62             } else {
 63                 // Found
 64                 return mid;
 65             }
 66         }
 67         // Not found
 68         return min;
 69     };

 71 	/**
 72 	 *  Provides an object containing a sorted array, and providing elementary
 73 	 *  set operations: merge (union), intersect, and substract (complement).
 74 	 *  It maintains internally a sorted array of data. The optional comparator
 75 	 *  must be an object of the form {lower_than: func_lt, equals: func_equal}
 76 	 *  
 77 	 *  @constructor
 78 	 *  @param comparatorObject a comparator object that provides two functions: lower_than and equals.
 79 	 *  If not defined, a default comparator using the javascript < operator is used. If you're only 
 80 	 *  storing integer values, you can safely omit this argument.
 81 	 */
 82 	fullproof.ResultSet = function(comparatorObject, data) {
 83 		if (!(this instanceof fullproof.ResultSet)) {
 84 			return new fullproof.ResultSet(comparatorObject,data);
 85 		}
 86 		this.comparatorObject = comparatorObject||defaultComparator;
 87 		this.data = data||[];
 88 		this.last_insert = undefined;
 89 	};

 91 	/**
 92 	 * Insert values into the array of data managed by this ResultSet. The insertion is optimized 
 93 	 * when the inserted values are greater than the last inserted values (the value is just pushed
 94 	 * to the end of the array). Otherwise, a binary search is done in the array to find the correct
 95 	 * offset where to insert the value. When possible, always insert sorted data.
 96 	 * 
 97 	 * @param values... any number of values to insert in this resultset.
 98 	 */
 99 	fullproof.ResultSet.prototype.insert = function() {
100 		for (var i=0; i<arguments.length; ++i) {
101 			var obj = arguments[i];
102 			
103 			if (this.last_insert && this.comparatorObject.lower_than(this.last_insert,obj)) {
104 				this.data.push(obj);
105 				this.last_insert = obj
106 			} else {
107 				var index = fullproof.binary_search(this.data, obj, undefined, undefined, this.comparatorObject.lower_than);
108 				if (index >= this.data.length) {
109 					this.data.push(obj);
110 					this.last_insert = obj
111 				} else if (this.comparatorObject.equals(obj, this.data[index]) == false) {
112 					this.data.splice(index, 0, arguments[i]);
113 					this.last_insert = undefined;
114 				}
115 			}
116 		}
117 		return this;
118 	};

120 	function defaultMergeFn(a,b) {
121 		return a;
122 	}
123 	
124 	/**
125      * Union operation. Merge another ResultSet or a sorted javascript array into this ResultSet.
126      * If the same value exists in both sets, it is not injected in the current set, to avoid duplicate values.
127      * @param set another ResultSet, or an array of sorted values
128      * @return this ResultSet, possibly modified by the merge operation
129      */
130     fullproof.ResultSet.prototype.merge = function (set, mergeFn) {
131         mergeFn = mergeFn || defaultMergeFn;
132         this.last_insert = undefined;
133         var other = false;
134 
135         if (set.constructor == Array) {
136             other = set;
137         } else if (set instanceof fullproof.ResultSet) {
138             other = set.getDataUnsafe();
139         }
140 
141 
142         var i1 = 0, max1 = this.data.length,
143             i2 = 0, max2 = other.length,
144             obj1 = null, obj2 = null;
145         var comp = this.comparatorObject;
146 
147         var result = [];
148         while (i1 < max1 && i2 < max2) {
149             obj1 = this.data[i1];
150             obj2 = other[i2];
151             if (comp.equals(obj1, obj2)) {
152                 result.push(mergeFn(obj1, obj2));
153                 ++i1;
154                 ++i2;
155             } else if (comp.lower_than(obj1, obj2)) {
156                 result.push(obj1);
157                 ++i1;
158             } else {
159                 result.push(obj2);
160                 ++i2;
161             }
162         }
163         while (i1 < max1) {
164             result.push(this.data[i1]);
165             ++i1;
166         }
167         while (i2 < max2) {
168             result.push(other[i2]);
169             ++i2;
170         }
171         this.data = result;
172         return this;
173     };

176 	/**
177      * Intersect operation. Modify the current ResultSet so that is only contain values that are also contained by another ResultSet or array.
178      * @param set another ResultSet, or an array of sorted values
179      * @return this ResultSet, possibly modified by the intersect operation
180      */
181     fullproof.ResultSet.prototype.intersect = function (set) {
182         this.last_insert = undefined;
183         var other = false;
184         if (set.constructor == Array) {
185             other = set;
186         } else if (set instanceof fullproof.ResultSet) {
187             other = set.getDataUnsafe();
188         }
189 
190         if (other) {
191             var result = [];
192             var i = 0, j = 0, maxi = this.data.length, maxj = other.length;
193             while (i < maxi) {
194                 while (j < maxj && this.comparatorObject.lower_than(other[j], this.data[i])) {
195                     ++j;
196                 }
197                 if (j < maxj && this.comparatorObject.equals(other[j], this.data[i])) {
198                     result.push(other[j]);
199                     ++i;
200                     ++j;
201                 } else {
202                     i++;
203                 }
204             }
205             this.data = result;
206         } else {
207             this.data = [];
208         }
209         return this;
210     };

213 	/**
214 	 * Substraction operation. Modify the current ResultSet so that any value contained in the provided set of values are removed.
215 	 * @param set another ResultSet, or an array of sorted values
216 	 * @return this ResultSet, possibly modified by the substract operation
217 	 */
218 	fullproof.ResultSet.prototype.substract = function(set) {
219 		this.last_insert = undefined;
220 		var other = false;
221 		if (set.constructor == Array) {
222 			other = set;
223 		} else if (set instanceof fullproof.ResultSet) {
224 			other = set.getDataUnsafe();
225 		}
226 		
227 		if (other) {
228 			var result = [];
229 			var i=0,j=0,maxi=this.data.length,maxj=other.length;
230 			while (i<maxi) {
231 				while (j<maxj && this.comparatorObject.lower_than(other[j],this.data[i])) {
232 					++j;
233 				}
234 				if (j<maxj && this.comparatorObject.equals(other[j],this.data[i])) {
235 					++i; 
236 					++j;
237 				} else {
238 					result.push(this.data[i]);
239 					i++;
240 				}
241 			}
242 			this.data = result;
243 		} else {
244 			this.data = [];
245 		}
246 		
247 		return this;
248 	};

277 	/**
278 	 * Changes the comparatorObject associated to this set, and sorts the data.
279 	 * Use this function if you want to sort the data differently at some point.
280 	 * @param comparatorObject the comparator to use
281 	 * @return this ResultSet instance
282 	 */
283 	fullproof.ResultSet.prototype.setComparatorObject = function(comparatorObject) {
284 		this.comparatorObject = comparatorObject;
285 		var self = this;
286 		this.data.sort(function(a,b) {
287 			if (self.comparatorObject.lower_than(a,b)) {
288 				return -1;
289 			} else if (self.comparatorObject.equals(a,b)) {
290 				return 0;
291 			} else {
292 				return 1;
293 			}
294 		});
295 	};

305 	/**
306      * Iterates over all the element of the array, and calls the provided function with each values.
307      * @param callback the function called with each element of the array
308      * @return this ResultSet instance
309      */
310     fullproof.ResultSet.prototype.forEach = function (callback) {
311         for (var i = 0, max = this.data.length; i < max; ++i) {
312             callback(this.data[i]);
313         }
314         return this;
