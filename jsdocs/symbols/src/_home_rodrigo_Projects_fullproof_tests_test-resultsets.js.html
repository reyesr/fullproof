  4 function CustomObject(id) {
  5 	if (!(this instanceof CustomObject)) {
  6 		return new CustomObject(id);
  7 	}
  8 	this.id = parseInt(id);
  9 	this.toString = function() {
 10 		return id.toString();
 11 	}
 12 }
 13 
 14 var CO = CustomObject;
 15 var COMP = {
 16 		lower_than: function(a,b) {
 17 			return parseInt(a.id) < parseInt(b.id);
 18 		},
 19 		equals: function(a,b) {
 20 			return a.id == b.id;
 21 		}
 22 	};
 23 
 24 function buildResultSetOfCustomObject(size, func) {
 25 	var result=new fullproof.ResultSet(COMP);
 26 	for (var i=0; i<size; ++i) {
 27 		var value = func?func(i):i;
 28 		result.insert(new CustomObject(value));
 29 	}
 30 	return result;
 31 }

 33 function verifyResultSetsOfCustomOject(rs1,rs2) {
 34 	var arr1 = (rs1 instanceof fullproof.ResultSet)?rs1.getDataUnsafe():rs1;
 35 	var arr2 = (rs2 instanceof fullproof.ResultSet)?rs2.getDataUnsafe():rs2;
 36 	equal(arr1.length,arr2.length);
 37 	for (var i=0; i<arr1.length && i<arr2.length; ++i) {
 38 		equal(arr1[i].id, arr2[i].id);
 39 	}
 40 }
